BeC3-lib

A very easy C++ Library to interact with objects connected on Bec3.

int main(int argc, const char **argv) {
    Bec3 mySession("assets/conf/Bec3.json");
    while(done){
    	mySession.updateObjects();
    	
    	if( mySession.getObjectState( "My" ).getBool() == true ){
    		/*Do something*/
    	}
    }
    return 0;
}
